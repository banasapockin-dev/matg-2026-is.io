<script>
let trigPoints = 0;
let logPoints = 0;

const trigQuestions = {
easy: {q:"sin(30°) = ?", a:["0.5","1","0"], c:0},
medium: {q:"cos(60°) = ?", a:["0.5","1","0"], c:0},
hard: {q:"tan(45°) = ?", a:["0","1","√2"], c:1}
};

const logQuestions = {
easy: {q:"log₁₀(100) = ?", a:["1","2","3"], c:1},
medium: {q:"log₂(8) = ?", a:["2","3","4"], c:1},
hard: {q:"log₃(27) = ?", a:["2","3","4"], c:1}
};

function showSection(id){
document.querySelectorAll("section").forEach(s=>s.classList.remove("active"));
document.getElementById(id).classList.add("active");
}

function startTrig(level){
let data = trigQuestions[level];
let html = `<h3>${data.q}</h3>`;
data.a.forEach((ans,i)=>{
html+=`<button class="answer" onclick="checkTrig(${i},${data.c})">${ans}</button>`;
});
document.getElementById("trigGame").innerHTML=html;
}

function startLog(level){
let data = logQuestions[level];
let html = `<h3>${data.q}</h3>`;
data.a.forEach((ans,i)=>{
html+=`<button class="answer" onclick="checkLog(${i},${data.c})">${ans}</button>`;
});
document.getElementById("logGame").innerHTML=html;
}

function checkTrig(ans,correct){
if(ans===correct){trigPoints+=10;alert("✅ Правильно! +10 очков");}
else{alert("❌ Неправильно");}
document.getElementById("trigScore").innerText="Очки: "+trigPoints;
}

function checkLog(ans,correct){
if(ans===correct){logPoints+=10;alert("✅ Правильно! +10 очков");}
else{alert("❌ Неправильно");}
document.getElementById("logScore").innerText="Очки: "+logPoints;
}
</script>
